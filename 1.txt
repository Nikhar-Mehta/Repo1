class person:
  def __init__(self):
    self.first_name = ''
    self.last_name = ''
    self.phone_number = ''
    self.uname = ''
    self.major = ''
  def set_first_name(self, first_name):
    self.first_name = first_name
  def get_first_name(self):
    return self.first_name
  
  def set_last_name(self, last_name):
    self.last_name = last_name
  def get_last_name(self):
    return self.last_name

  def set_phone_number(self, phone_number):
    self.phone_number = phone_number
  def get_phone_number(self):
    return self.phone_number

  def set_uname(self, uname):
    self.uname = uname
  def get_uname(self):
    return self.uname

  def set_major(self, major):
    self.major = major
  def get_major(self):
    return self.major

  def get_all(self):
    diction = {}
    diction["Fname"] = self.first_name
    diction["Lname"] = self.last_name
    diction["number"] = self.phone_number
    diction["uname"] = self.uname
    diction["Major"] = self.major
    return diction
